package root.pow;

/**
 * @author Victor Kuliamin
 */
public class Power {

  public int pow(int a, int b) {
    int r = 1;

    while (b > 0) {
      if ((b & 1) != 0)

        r *= a;

      a *= a;
      b >>= 1;
    }

    return r % Integer.MAX_VALUE;
  }
}
